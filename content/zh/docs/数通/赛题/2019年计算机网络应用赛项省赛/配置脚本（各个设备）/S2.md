```
# 设备命名
hostname GZXQ-S2910-02

# 配置SSH
no password policy strong
no password policy min-size 
no password policy forced-password-modify

no service password-encryption

no enable secret
enable password 0 admin

username admin privilege 1 password 0 admin
username admin login mode ssh

enable service ssh-server
line vty 0 4
login local
transport input ssh
exit

# 创建VLAN并重命名VLAN
vlan range 10,20,30,100
exit

vlan 10
name Jiaoxue
exit
vlan 20
name Sushe
exit
vlan 30
name Tushu
exit
vlan 100
name Manage
exit

# 配置VLAN地址和回环地址（无回环地址）
interface VLAN 100
ip address 192.1.100.2 24
exit

# 配置VLAN接口类型
interface range gigabitEthernet 0/1 - 4
switchport mode access
switchport access vlan 10
exit
interface range gigabitEthernet 0/5 - 8
switchport mode access
switchport access vlan 20
exit
interface range gigabitEthernet 0/9 - 11
switchport mode access
switchport access vlan 30
exit

interface range gigabitEthernet 0/23 - 24
switchport mode trunk
switchport trunk allowed vlan only 10,20,30,100
exit

# 配置MSTP
spanning-tree
spanning-tree mode mstp
spanning-tree mst configuration
instance 1 vlan 50,60,70
instance 2 vlan 10,20,30,100
revision 1
name ruijie
exit

# 配置RLDP防环、BPDU保护、边缘端口
rldp enable

interface range gigabitEthernet 0/1 - 9
rldp port loop-detect shutdown-port
spanning-tree portfast
 
errdisable recovery interval 300
exit

# 开启DHCP snooping和关闭NFPP的ARP检测
ip dhcp snooping

interface range gigabitEthernet 0/23 - 24
ip dhcp snooping trust
no nfpp arp-guard enable
exit

# 配置NFPP日志缓存和log打印阈值
nfpp
log-buffer entries 1024
log-buffer logs 1 interval 300
exit

# 配置CPU保护机制阈值为500pps
cpu-protect cpu bandwidth 500

# 配置端口安全, 接口g0/9只允许PC2通过
interface gigabitEthernet 0/9
switchport port-security
switchport port-security maximum 1
switchport port-security mac-address e0d4.64e0.bcad
exit

```
