```
# AC1

vlan range 10,20,30
exit
vlan 10
name User
exit
vlan 20
name Wire_user
exit
vlan 30
name AP
exit

interface vlan 10
ip address 195.1.10.254 255.255.255.0
exit
interface vlan 20
ip address 195.1.20.254 255.255.255.0
exit
interface vlan 30
ip address 195.1.30.254 255.255.255.0
exit

interface gigabitEthernet 0/1
no switchport
ip address 10.1.0.1 255.255.255.252
exit
interface loopback 0
ip address 11.1.0.204 255.255.255.255
exit
interface gigabitEthernet 0/5
switchport mode trunk
switchport trunk native vlan 30
switchport trunk allowed vlan only 20,30
exit

interface range gigabitEthernet 0/2-4
switchport mode access
switchport access vlan 10
exit

```
